%% SLIDE 4 — Introducing Datacentering
\begin{frame}{Introducing ``Datacentering''}

\metroset{block=fill}
\begin{block}{Concept}
  \textbf{Datacentering} = the active socio-material process through which digital
  infrastructure is continuously assembled, contested, and reproduced
\end{block}

\vspace{0.4em}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
  \textbf{Why a gerund?}\small
  \begin{itemize}
    \item Process over static object
    \item Who is doing the datacentering?
    \item What is assembled or foreclosed?
  \end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
  \textbf{Theoretical grounding}\small
  \begin{itemize}
    \item Orlikowski (2007): sociomaterial entanglement
    \item Jasanoff \& Kim (2015): imaginaries co-produce facts
    \item ``Cloud'' suppresses what datacentering names
  \end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{A Multimodal Research Agenda}

\vspace{-0.2em}
\begin{center}
\small
\begin{tabularx}{0.97\textwidth}{>{\bfseries}lllX}
\toprule
WP & CAS Dimension & Data & Output \\

\vspace{0.4em}
\metroset{block=fill}
\begin{block}{Why multimodal?}
\small Strategic opacity is a \textit{sociotechnical fact} --- not a data gap.
A firm that won't publish is still in a REIT filing, a satellite image, and a planning record.
\end{block}

\end{frame}

\begin{frame}{Building the Cartography Together}
\begin{columns}[T]
\begin{column}{0.54\textwidth}
  \small\textbf{Why open exchange is a theoretical necessity}\\[0.3em]
  Datacentering is CAS-like: too large, too distributed, too opaque for any single team.
  The research commons is the methodological response.\\[0.3em]
  \textbf{What we share:}\small
  \begin{itemize}
    \item Community facility index (CC BY 4.0)
    \item NLP / CV / network pipelines (MIT)
    \item Open data and open workflows
  \end{itemize}
\end{column}
\begin{column}{0.42\textwidth}
  \small\textbf{Concrete asks:}
  \begin{itemize}
    \item Local knowledge: disputes, unreported facilities
    \item Company registry or REIT data access
    \item Methodological critique
    \item Replication in other languages
  \end{itemize}
\end{column}
\end{columns}
