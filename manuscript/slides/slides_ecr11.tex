%% SLIDE 5 — Theory: Arenas  0:50
\begin{frame}{\iconfull{dcblue}{users}\enspace Theory: Arenas of Transformation}
\begin{columns}[T]
\begin{column}{0.52\textwidth}
  \small\textbf{After \href{https://doi.org/10.1016/j.respol.2012.03.001}{Jørgensen's} \textit{arenas of development} (2012, \textit{Res.\ Policy})}\\[0.3em]
  \footnotesize
  Flat ontology. Start from \textit{actor\\
  constellations} and their collective\\
  engagement — not structural tiers.\\[0.4em]
  \textbf{Negotiation} is the fundamental process.\\[0.4em]
  Goal: \textbf{characterize what is \textit{emerging}}\\
  before the arena stabilises.\\[0.5em]
  \small\textbf{+ Clarke (2005): situational maps}\\[0.3em]
  \footnotesize
  Implicated actors: silenced (communities,\\
  visa workers) + constructed but absent\\
  (``the consumer'', ``future generations'')
\end{column}
\begin{column}{0.44\textwidth}
  \vspace{0.3em}
  \footnotesize\textbf{The datacentering arena (in formation):}\\[0.4em]
  \begin{tikzpicture}[font=\scriptsize,node distance=0.2cm,
    actor/.style={draw=black!30,rounded corners,inner sep=3pt,fill=black!4}]
    \node[actor] (h)  {Hyperscalers};
    \node[actor,below=of h] (g) {Grid operators};
    \node[actor,below=of g] (c) {\textcolor{dchighlight}{Communities}};
    \node[actor,right=0.5cm of h] (w)  {Workers};
    \node[actor,below=of w] (r)  {Regulators};
    \node[actor,below=of r] (p)  {Planners};
    \node[draw=dchighlight,dashed,rounded corners,
          fit=(h)(g)(c)(w)(r)(p),
          label=above:{\textcolor{dchighlight}{\textbf{arena}}}]{};
  \end{tikzpicture}
  \vspace{0.3em}

  \footnotesize Co-constitutive.\\
  Not hierarchically ordered.
\end{column}
\end{columns}
\note{Arenas of Transformation — not Development. Development implies something already underway with a known trajectory. Transformation asks: what is this arena doing to the systems around it? That's the empirical question. Jørgensen is in RP 2012 — your canon. We adapt, not import wholesale. Clarke gives us implicated actors — who is excluded from the negotiation.}
\end{frame}

%% APPENDIX — Tensions & Paradoxes × Transitions Agenda
\begin{frame}[noframenumbering,allowframebreaks]{%
  \faIcon{exclamation-triangle}\enspace Datacentering × Transitions Research Agenda}
\vspace{0.2em}
{\scriptsize Tensions and paradoxes surfaced by the \textit{arena of transformation} framing,
mapped onto the nine themes of the updated transitions research agenda
(Geels et al.\ 2024, \textit{Environ.\ Innov.\ Soc.\ Transitions}).}
\vspace{0.5em}

\begin{tabularx}{\linewidth}{@{}p{3.4cm}X@{}}
\toprule
\textbf{\scriptsize Theme} & \textbf{\scriptsize Tension / Paradox in Datacentering} \\
\midrule
\scriptsize\textbf{i.\ Understanding transitions}
\newline\textcolor{dcgray}{\tiny Geels, Fünfschilling}
&
\tiny DCs simultaneously \textit{enable} digital transitions and \textit{block} energy transitions. Same infrastructure; opposite systemic effects. Standard transition frameworks assume one trajectory — datacentering has at least two. \\[0.3em]

\scriptsize\textbf{ii.\ Power \& politics}
\newline\textcolor{dcgray}{\tiny Avelino, Kern}
&
\tiny Hyperscalers negotiate bilaterally with grid operators at national scale yet face municipal veto at planning level. Power is scale-incommensurable — no single arena contains it. \\[0.3em]

\scriptsize\textbf{iii.\ Governing transitions}
\newline\textcolor{dcgray}{\tiny Nykvist, Kivimaa}
&
\tiny AI infrastructure strategy formed nationally (Stargate, EU AI Act); planning permission granted locally. Governance most fragmented precisely where infrastructure is most cross-scalar. \\[0.3em]

\scriptsize\textbf{iv.\ Civil society \& movements}
\newline\textcolor{dcgray}{\tiny Hess, Rohracher}
&
\tiny ``Clean cloud'' is marketed globally; noise, heat-island, and water stress are locally borne. Spatial mismatch of benefit and burden is structural, not incidental. \\[0.3em]

\scriptsize\textbf{v.\ Organizations \& industries}
\newline\textcolor{dcgray}{\tiny Markard, Wells}
&
\tiny Three hyperscalers control $>$50\% of global capacity while claiming to enable a distributed, competitive digital economy. Sector consolidation and democratisation discourse co-evolve. \\[0.3em]

\scriptsize\textbf{vi.\ Practice \& everyday life}
\newline\textcolor{dcgray}{\tiny McMeekin, Hyysalo et al.}
&
\tiny ``Dematerialisation'' (cloud = less hardware) coexists with the fastest physical infrastructure build on record. Jevons paradox at the cultural level: invisibility amplifies consumption. \\[0.3em]

\scriptsize\textbf{vii.\ Geography of transitions}
\newline\textcolor{dcgray}{\tiny Raven, Wieczorek}
&
\tiny Global cloud routes through three postcodes — Ashburn, AMS-IX, Slough. Geographic concentration \textit{increases} as scale grows; ``the cloud'' is the most place-specific infrastructure since the railway. \\[0.3em]

\scriptsize\textbf{viii.\ Justice \& ethics}
\newline\textcolor{dcgray}{\tiny Sovacool, Jenkins, Onsongo}
&
\tiny Efficiency gains (liquid cooling, renewable PPAs) absorbed by demand growth. Carbon-neutral pledges coexist with the largest single-sector electricity surge in a generation. \\[0.3em]

\scriptsize\textbf{ix.\ Methodologies}
\newline\textcolor{dcgray}{\tiny Boons, Alkemade, Holtz, Pel}
&
\tiny Case study misses scale; computational text mining misses political texture. And the infrastructure we study mediates the trade-press data we use to study it — reflexivity problem. \\
\bottomrule
\end{tabularx}
\end{frame}

\begin{frame}[noframenumbering]{\faIcon{question-circle}\enspace Q: Why CITC and not CSS more broadly?}
\footnotesize
CITC's lexical-framing requirement is what makes the output theoretical, not just descriptive.\\[0.5em]
That's the discipline that protects against the most common reviewer objection:\\[0.3em]
\begin{quote}
  ``This is interesting, but what's the theoretical contribution?''
\end{quote}
\vspace{0.3em}
CITC explicitly distinguishes patterns from contributions. The test for RP.
\end{frame}
